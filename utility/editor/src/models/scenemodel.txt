import { DirectionalLight, PerspectiveCamera, Scene } from '@zephyr3d/scene';
import { BaseModel } from './basemodel';
import { Vector3 } from '@zephyr3d/base';
import { EditorCameraController } from '../helpers/editorcontroller';

export class SceneModel extends BaseModel {
  private _scene: Scene;
  private _getTargetCenter: () => Vector3 | null; // 新增：用於儲存回調函數的參考

  constructor(getTargetCenter?: () => Vector3 | null) { // 修改建構函數
    super();
    this._scene = null;
    // 如果有傳入回調，使用它；否則使用總是返回 null 的預設函數
    this._getTargetCenter = getTargetCenter || (() => null);
  }

  // 新增：允許動態設置回調函數
  setTargetCenterCallback(callback: () => Vector3 | null): void {
    this._getTargetCenter = callback;
    
    // 關鍵修復：更新現有攝像機控制器的回調
    if (this._scene?.mainCamera?.controller instanceof EditorCameraController) {
      (this._scene.mainCamera.controller as EditorCameraController).setTargetCenterCallback(callback);
    }
  }

  get scene() {
    return this._scene;
  }

  reset(scene?: Scene) {
    this._scene?.dispose();
    this._scene = scene ?? new Scene();
    if (!scene) {
      this._scene.env.sky.fogType = 'height_fog';
      const light = new DirectionalLight(this._scene);
      light.lookAt(Vector3.one(), Vector3.zero(), Vector3.axisPY());
    }
    if (!this._scene.mainCamera) {
      const defaultCamera = new PerspectiveCamera(this._scene, Math.PI / 3, 1, 1000);
      defaultCamera.lookAt(new Vector3(0, 8, 18), new Vector3(0, 0, 0), Vector3.axisPY());
      this._scene.mainCamera = defaultCamera;
    }
    
    // 修改：先創建控制器，然後設置回調
    const controller = new EditorCameraController(
      undefined, // 使用預設選項
      () => this._getTargetCenter() // 直接代理到当前的回调函数引用
    );
    this._scene.mainCamera.controller = controller;
  }
}
